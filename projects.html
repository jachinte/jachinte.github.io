              package co.migueljimenez.app.monitors
  monitor Latency {
    val target = $domain.child("Strassen").child("matrix").service("multiplication")
    event e raised on return of target

    handler onReturn(ReturnEvent e) {
      val tags = #{
        "strategy" -> "strassen",
        "host" -> "grid0",
        "component" -> "Strassen"
      }
      State.latency = tag(e.value, tags)
    }

    config {
      e.bindingUri = new URI("http://grid0:" + 3000)
      e.subscribe(onReturn)
    }
  }

              package co.edu.icesi.driso.matrices

  import java.util.List
  import java.util.Map
  import org.amelia.dsl.lib.descriptors.Host
  import org.amelia.dsl.lib.util.Hosts

  subsystem Common {
      val String artifact = "mcm-common"

      // The folder containing all of the Java projects
      param String sources = "/tmp/matrices"

      // The folder containing the compiled jar files
      param String assets = "assets"

      // Common dependencies
      param List classpath = #['/.jar'];

      // All hosts, organized by host identifier
      param Map hosts = Hosts.hosts("hosts.txt").toMap[h|h.identifier]

      on hosts.values {
              init:
                cd sources
                compile "common/src" '/'
      }
  }
